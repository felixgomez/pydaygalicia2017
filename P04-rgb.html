gpiozero - RGBLED 

#!/usr/bin/env python3

from gpiozero import RGBLED
from time import sleep

rgb = RGBLED(5, 6, 13)

while True:
    rgb.color = (1, 0, 0) # Tupla
    sleep(0.5)
    rgb.color = (0, 1, 0) # Tupla
    sleep(0.5)
    rgb.color = (0, 0, 1) # Tupla
    sleep(0.5)

Código Python

Recorriendo rangos


#!/usr/bin/env python3

from gpiozero import RGBLED
from time import sleep

rgb = RGBLED(5, 6, 13)

while True:
	for i in [x * 0.05 for x in range(0, 10)]:
		sleep(0.1)
		rgb.color = (i,0,0)
